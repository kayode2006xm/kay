                {x: 550, o: 95, h: 115, l: 90, c: 110, color: 'green'},
                {x: 600, o: 110, h: 130, l: 105, c: 125, color: 'green'},
                {x: 650, o: 125, h: 140, l: 120, c: 135, color: 'green'},
                {x: 700, o: 135, h: 150, l: 130, c: 140, color: 'green'},
                {x: 750, o: 140, h: 160, l: 135, c: 145, color: 'green'},
            ];
            
            // Scale candles to chart height
            const scaleCandles = candles.map(c => ({
                x: c.x,
                o: height - (c.o * 2.5),
                h: height - (c.h * 2.5),
                l: height - (c.l * 2.5),
                c: height - (c.c * 2.5),
                color: c.color
            }));
            
            // Draw candlesticks
            scaleCandles.forEach(candle => {
                // Wick
                ctx.strokeStyle = candle.color === 'green' ? '#10b981' : '#ef4444';
                ctx.lineWidth = 1;
                ctx.beginPath();
                ctx.moveTo(candle.x, candle.h);
                ctx.lineTo(candle.x, candle.l);
                ctx.stroke();
                
                // Body
                ctx.fillStyle = candle.color === 'green' ? '#10b981' : '#ef4444';
                ctx.fillRect(candle.x - 6, candle.o, 12, candle.c - candle.o);
                ctx.strokeStyle = candle.color === 'green' ? '#059669' : '#dc2626';
                ctx.strokeRect(candle.x - 6, candle.o, 12, candle.c - candle.o);
            });
            
            // Draw trendlines
            ctx.strokeStyle = '#3b82f6';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(50, scaleCandles[0].l);
            ctx.lineTo(350, scaleCandles[6].l);
            ctx.stroke();
            
            ctx.strokeStyle = '#8b5cf6';
            ctx.beginPath();
            ctx.moveTo(200, scaleCandles[3].h);
            ctx.lineTo(500, scaleCandles[9].h);
            ctx.stroke();
            
            // Draw supply zone
            ctx.fillStyle = 'rgba(239, 68, 68, 0.2)';
            ctx.beginPath();
            ctx.moveTo(400, scaleCandles[7].h - 20);
            ctx.lineTo(700, scaleCandles[7].h - 20);
            ctx.lineTo(700, scaleCandles[7].h + 40);
            ctx.lineTo(400, scaleCandles[7].h + 40);
            ctx.closePath();
            ctx.fill();
            ctx.strokeStyle = 'rgba(239, 68, 68, 0.5)';
            ctx.stroke();
            
            // Draw demand zone
            ctx.fillStyle = 'rgba(16, 185, 129, 0.2)';
            ctx.beginPath();
            ctx.moveTo(100, scaleCandles[1].l + 20);
            ctx.lineTo(250, scaleCandles[1].l + 20);
            ctx.lineTo(250, scaleCandles[1].l - 40);
            ctx.lineTo(100, scaleCandles[1].l - 40);
            ctx.closePath();
            ctx.fill();
            ctx.strokeStyle = 'rgba(16, 185, 129, 0.5)';
            ctx.stroke();
            
            // User menu dropdown
            const userMenu = document.getElementById('user-menu');
            const menuDropdown = document.getElementById('menu-dropdown');
            
            userMenu.addEventListener('click', function() {
                menuDropdown.classList.toggle('hidden');
            });
            
            // Close dropdown when clicking outside
            document.addEventListener('click', function(event) {
                if (!userMenu.contains(event.target) && !menuDropdown.contains(event.target)) {
                    menuDropdown.classList.add('hidden');
                }
            });
        });
